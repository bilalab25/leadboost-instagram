### 3️⃣ Cambios en `storage`
Agrega estos métodos (puedes implementarlos con Prisma, Supabase o tu capa actual):

```js
async function bulkInsertMessages(userId, integrationId, messages) {
  for (const msg of messages) {
    await db.messages.upsert({
      where: { metaMessageId: msg.messageId },
      update: {},
      create: {
        userId,
        integrationId,
        platform: msg.provider,
        metaMessageId: msg.messageId,
        senderId: msg.senderId,
        recipientId: msg.recipientId,
        textContent: msg.textContent,
        direction: msg.direction,
        isRead: false,
        timestamp: new Date(msg.created_time),
      },
    });
  }
}

async function markIntegrationAsFetched(integrationId) {
  await db.integrations.update({
    where: { id: integrationId },
    data: { hasFetchedHistory: true },
  });
}

async function findIntegrationByAccount(accountId, provider) {
  return await db.integrations.findFirst({
    where: { account_id: accountId, provider },
  });
}
```
